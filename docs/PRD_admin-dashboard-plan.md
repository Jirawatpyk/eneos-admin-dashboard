### 5.3 Authentication Middleware

```typescript
// middleware/admin-auth.ts
import { Request, Response, NextFunction } from 'express';
import { verifyGoogleToken } from '../lib/google-auth.js';

const ALLOWED_DOMAINS = ['eneos.co.th'];
const ADMIN_EMAILS = [
  'manager@eneos.co.th',
  'director@eneos.co.th',
  'md@eneos.co.th'
];

export async function adminAuthMiddleware(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const authHeader = req.headers.authorization;

  if (!authHeader?.startsWith('Bearer ')) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  const token = authHeader.split(' ')[1];

  try {
    const payload = await verifyGoogleToken(token);
    const email = payload.email;

    // Check domain
    const domain = email.split('@')[1];
    if (!ALLOWED_DOMAINS.includes(domain)) {
      return res.status(403).json({ error: 'Forbidden - Invalid domain' });
    }

    // Optional: Check specific admin emails
    // if (!ADMIN_EMAILS.includes(email)) {
    //   return res.status(403).json({ error: 'Forbidden - Not an admin' });
    // }

    req.user = { email, name: payload.name };
    next();
  } catch (error) {
    return res.status(401).json({ error: 'Invalid token' });
  }
}
```

## 8. Security Considerations

### 8.1 Authentication
- Google OAuth 2.0 with domain restriction
- Session-based authentication (NextAuth.js)
- HTTPS only

### 8.2 Authorization
- Domain whitelist: `@eneos.co.th`
- Optional: Admin email whitelist
- API token validation on backend

### 8.3 Data Protection
- No sensitive data in client-side storage
- API responses filtered (no passwords, secrets)
- Rate limiting on API endpoints
