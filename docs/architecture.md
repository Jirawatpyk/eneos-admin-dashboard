### 11.3 Component Responsive Behavior

| Component | Desktop (lg+) | Tablet (md) | Mobile (base) |
|-----------|---------------|-------------|---------------|
| Sidebar | Fixed 256px | Sheet/Drawer | Sheet/Drawer |
| Header | Full nav | Hamburger | Hamburger |
| KPI Grid | 4 columns | 2 columns | 1 column |
| Charts | Side-by-side | Stacked | Stacked |
| Tables | Full width | Horizontal scroll | Card view option |
| Modals | 600px max | 90% width | Full screen |
| Filters | Inline | Collapsible | Drawer |

### 11.4 Responsive Component Pattern

```typescript
// Example: Responsive Dashboard Layout
export function DashboardLayout({ children }) {
  return (
    <div className="flex min-h-screen">
      {/* Desktop Sidebar - Hidden on mobile */}
      <aside className="hidden lg:block lg:w-64 lg:fixed lg:inset-y-0">
        <Sidebar />
      </aside>

      {/* Mobile Header with Hamburger */}
      <div className="lg:hidden fixed top-0 w-full z-50">
        <MobileHeader />
      </div>

      {/* Main Content - Responsive padding */}
      <main className="flex-1 lg:pl-64 pt-16 lg:pt-0">
        <div className="px-4 sm:px-6 lg:px-8 py-6">
          {children}
        </div>
      </main>

      {/* Mobile Navigation Sheet */}
      <Sheet>
        <MobileNav />
      </Sheet>
    </div>
  );
}

// Example: Responsive KPI Grid
export function KPIGrid({ items }) {
  return (
    <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-4">
      {items.map(item => (
        <KPICard key={item.id} {...item} />
      ))}
    </div>
  );
}

// Example: Responsive Table with Card Fallback
export function ResponsiveTable({ data }) {
  return (
    <>
      {/* Desktop Table */}
      <div className="hidden md:block overflow-x-auto">
        <table className="min-w-full">...</table>
      </div>

      {/* Mobile Card View */}
      <div className="md:hidden space-y-4">
        {data.map(item => (
          <MobileCard key={item.id} {...item} />
        ))}
      </div>
    </>
  );
}
```
